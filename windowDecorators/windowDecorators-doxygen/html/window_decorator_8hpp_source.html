158 class combine_windows : public window {
159  private:
160  array<window*, N> windows;
161  int num_rows, num_columns;
162  bool uniform_size;
163 
170  void write_implementation(xy pos, color col) override {
171  int final_row = 0;
172  int final_column = 0;
173  int current_size_x = 0;
174  int current_size_y = 0;
175  if (uniform_size) {
176  current_size_x = windows[0]->size.x;
177  current_size_y = windows[0]->size.y;
178  final_row = pos.y / current_size_y;
179  final_column = pos.x / current_size_x;
180  int final_x = pos.x % current_size_x;
181  int final_y = pos.y % current_size_y;
182  pos = xy(final_x, final_y);
183  } else {
184  bool found_x = false, found_y = false;
185  int check_index = 0;
186 
187  while (!found_x || !found_y) {
188  current_size_x = windows[check_index]->size.x;
189  current_size_y = windows[check_index]->size.y;
190 
191  if (pos.x > current_size_x - 1) {
192  final_column++;
193  pos = pos - xy(current_size_x, 0);
194  } else {
195  found_x = true;
196  }
197  if (pos.y > current_size_y - 1) {
198  final_row++;
199  pos = pos - xy(0, current_size_y);
200  } else {
201  found_y = true;
202  }
203  check_index++;
204  }
205  }
206  int final_screen = final_column + (final_row * num_columns);
207  windows[final_screen]->write(pos);
208  }
209 
210  public:
219  combine_windows(const array<window*, N> windows, int num_rows, int num_column, xy total_size,
220  bool uniform_size = false)
221  : window(total_size), windows(windows), num_rows(num_rows), num_columns(num_column), uniform_size(uniform_size) {}
222 
228  void clear() override {
229  for (unsigned int i = 0; i < N; i++) {
230  windows[i]->clear();
231  }
232  }
233 
239  void flush() override {
240  for (unsigned int i = 0; i < N; i++) {
241  windows[i]->flush();
242  }
243  }
244 };
