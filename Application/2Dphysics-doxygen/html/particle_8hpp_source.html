203  template <unsigned int N>
215  void updatePosition(const double& delta_time, const array<particle*, N>& particles, unsigned int id) {
216  float new_x = real_pos.x + (speed.x * delta_time); // New x-position of the particle
217  float new_y = real_pos.y + (speed.y * delta_time); // New y-position of the particle
218  // =================== Collision ==========================
219  for (unsigned int i = id + 1; i < N; i++) { // Only check for collisions with particles that are not checked yet
220  vector pos_comp = particles[i]->getRealPos();
221  // Calculate the distance between the two particles.
222  float distance = ((pos_comp.x - real_pos.x) * (pos_comp.x - real_pos.x)) +
223  ((pos_comp.y - real_pos.y) * (pos_comp.y - real_pos.y));
224  if (distance < 1) { // The 1 is the radius+radius squared
225  // If true than there is a collision, so position doesn't change
226  new_x = real_pos.x;
227  new_y = real_pos.y;
228  speed = {0.0, 0.0};
229  break;
230  }
231  }
232  if (new_x > x_max - radius) { // Checks if new_x is out of bounce
233  new_x = x_max - radius;
234  speed.x = 0;
235  } else if (new_x < 0 + radius) {
236  new_x = 0 + radius;
237  speed.x = 0;
238  }
239  if (new_y > y_max - radius) { // Checks if new_y is out of bounce
240  new_y = y_max - radius;
241  speed.y = 0;
242  } else if (new_y < 0 + radius) {
243  new_y = 0 + radius;
244  speed.y = 0;
245  }
246  real_pos.x = new_x; // No collision or out of bounce, so new_x is oke
247  real_pos.y = new_y; // No collision or out of bounce, so new_y is oke
248  }

302  void updateAcceleration() {
303  int angle_x = mpu.getAngleX();
304  int angle_y = mpu.getAngleY();
305  if (angle_x > accel_threshold || angle_x < -accel_threshold) {
306  acceleration.x = (max_acceleration * (angle_x / 90.0));
307  } else {
308  acceleration.x = 0;
309  }
310  if (angle_y > accel_threshold || angle_y < -accel_threshold) {
311  acceleration.y = -1 * (max_acceleration * (angle_y / 90.0));
312  } else {
313  acceleration.y = 0;
314  }
315  }

323  void updateParticles() {
324  uint64_t current_time = now_us();
325  double d_time = (current_time - previous_time) / 1e6; // current_time is in nanoseconds
326  previous_time = current_time;
327  for (unsigned int i = 0; i < N; i++) {
328  particles[i]->updateSpeed(acceleration, d_time, resis_const);
329  }
330  for (unsigned int i = 0; i < N; i++) {
331  particles[i]->updatePosition(d_time, particles, i);

333  }
